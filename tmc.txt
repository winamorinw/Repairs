        DAL dal = new DAL();
        public myfrmSearchTMC(string remusl, string ceh, string uth )
        {
            tip_remusl = remusl;
            cceh = ceh;
            cuth = uth;
            InitializeComponent();
            if (remusl == "rem")
            {
                dgv_AllTMC.DataSource = dal.GetAllTMC();
                button2.Visible = true;
                this.Text = "Выбор Основного средства";
                groupBox1.Text = "Список ОС";
            }

            if (remusl == "usl")
            {
                dgv_AllTMC.DataSource = dal.GetAllTMCUsl();
                button2.Visible = false;
                this.Text = "Выбор №";
                groupBox1.Text = "Список";
            }

            if (remusl == "1prostoi" || remusl == "5prostoi")
            {
                if (remusl == "1prostoi")
                dgv_AllTMC.DataSource = dal.GetAllTMC_Prostoi(cceh,cuth);
                
                if (remusl == "5prostoi")
                dgv_AllTMC.DataSource = dal.GetAllTMC_Prostoi(cceh,cuth);

                button2.Visible = true;
                this.Text = "Выбор Основного средства";
                groupBox1.Text = "Список ОС";
            }

            if (dgv_AllTMC.RowCount > 0)
            { SettingsAllOS(); }
            else
            {
                MessageBox.Show("У Вас нет ОС которые можно отправить в ремонт");
                this.Close();
            }
        }

        private void button1_Click(object sender, EventArgs e)
        {
            
            Inv_Number.Inv_n = dgv_AllTMC.CurrentRow.Cells["InventoryNumber"].Value.ToString();
            Inv_Number.Inv_name = dgv_AllTMC.CurrentRow.Cells["Name"].Value.ToString();
            Inv_Number.Obekt = dgv_AllTMC.CurrentRow.Cells["Obekt"].Value.ToString();
            CloseByButton = true;
            this.Close();
        }

        private void button2_Click(object sender, EventArgs e)
        {
            for (int i = 0; i < dgv_AllTMC.RowCount; i++)
            {
                dgv_AllTMC.Rows[i].Selected = false;
                for (int j = 0; j < dgv_AllTMC.ColumnCount; j++)
                    if (dgv_AllTMC.Rows[i].Cells[j].Value != null)
                        if (dgv_AllTMC.Rows[i].Cells[j].Value.ToString().Contains(txt_Search.Text))
                        {
                            dgv_AllTMC.Rows[i].DefaultCellStyle.BackColor = Color.Aqua;
                            break;
                        }
            }
        }
        //поиск по инвентарному номеру
        private void txt_Search_TextChanged(object sender, EventArgs e)
        {
            if (txt_Search.Text.Length > 2)
            {
                SearchForInventory();
            }
            else
            {
                if (tip_remusl == "rem")
                {
                    dgv_AllTMC.DataSource = dal.GetAllTMC();
                }
                if (tip_remusl == "usl")
                {
                   dgv_AllTMC.DataSource = dal.GetAllTMCUsl();
                }
                if (tip_remusl == "1prostoi" || tip_remusl == "5prostoi")
                {
                    if (tip_remusl == "1prostoi")
                        dgv_AllTMC.DataSource = dal.GetAllTMC_Prostoi(cceh,cuth);

                    if (tip_remusl == "5prostoi")
                        dgv_AllTMC.DataSource = dal.GetAllTMC_Prostoi(cceh,cuth);


                }
            }
        }


        private void txt_SearchName_TextChanged(object sender, EventArgs e)
        {
            if (txt_SearchName.Text.Length > 2)
            {
                SearchForName();
            }
            else
            {
                if (tip_remusl == "rem")
                {
                    dgv_AllTMC.DataSource = dal.GetAllTMC();
                }
                if (tip_remusl == "usl")
                {
                    dgv_AllTMC.DataSource = dal.GetAllTMCUsl();
                }
                if (tip_remusl == "1prostoi")
                {
                    dgv_AllTMC.DataSource = dal.GetAllTMC_Prostoi(cceh,cuth);
                }
                if (tip_remusl == "5prostoi")
                {
                    dgv_AllTMC.DataSource = dal.GetAllTMC_Prostoi(cceh,cuth);
                }

            }
        }

        //поиск по инвентарному номеру
        private void SearchForInventory()
        {
            dgv_AllTMC.ReadOnly = false;
            for (int i = 0; i < dgv_AllTMC.RowCount; i++)
            {
                if (dgv_AllTMC.Rows[i].Cells["InventoryNumber"].Value.ToString().Contains(txt_Search.Text) == false)
                {
                    dgv_AllTMC.CurrentCell = null;
                    dgv_AllTMC.Rows[i].Visible = false;
                }
            }
        }

        //поиск по названию
        private void SearchForName()
        {
            dgv_AllTMC.ReadOnly = false;
            for (int i = 0; i < dgv_AllTMC.RowCount; i++)
            {
                if (dgv_AllTMC.Rows[i].Cells["Name"].Value.ToString().ToLowerInvariant().Contains(txt_SearchName.Text) == false)
                {
                    dgv_AllTMC.CurrentCell = null;
                    dgv_AllTMC.Rows[i].Visible = false;
                }
            }

        }
